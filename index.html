// 讀取 Google 日曆事件並推入 googleEvents 陣列 async function fetchGoogleEvents(token) { try { showLoading(); const now = new Date(); // 抓取過去 2 個月到未來 1 年內的行程 const timeMin = new Date(now.getFullYear(), now.getMonth() - 2, 1).toISOString(); const timeMax = new Date(now.getFullYear() + 1, now.getMonth(), 1).toISOString(); const res = await fetch(`https://www.googleapis.com/calendar/v3/calendars/primary/events?timeMin=${timeMin}&timeMax=${timeMax}&singleEvents=true&orderBy=startTime`, { headers: { 'Authorization': `Bearer ${token}` } }); const data = await res.json(); // 如果 API 回傳錯誤，主動拋出並顯示原因 if (!res.ok) { throw new Error(data.error?.message || '未知的 API 錯誤'); } if (data.items) { googleEvents = data.items.map(item => { const start = item.start.dateTime || item.start.date; const end = item.end.dateTime || item.end.date; return { id: item.id, title: '📅 ' + (item.summary || '未命名行程'), startDate: start.split('T')[0], endDate: end.split('T')[0], startTime: item.start.dateTime ? start.split('T')[1].substring(0,5) : null, endTime: item.end.dateTime ? end.split('T')[1].substring(0,5) : null, location: item.location || '', description: item.description || '', color: '#34A853' // Google 專屬綠色標記 }; }); } hideLoading(); } catch(e) { console.error("Fetch Google Events failed:", e); hideLoading(); // 讓錯誤直接跳出警告視窗，不再默默沒反應 alert("Google 日曆同步失敗：\n" + e.message + "\n\n👉 請確認是否已在 Google Console 啟用「Google Calendar API」，並且在登入畫面有勾選「查看日曆」的權限！"); } }
